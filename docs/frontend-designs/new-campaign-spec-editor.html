# This campaign will visit all repositories in the organization and update the React version
name: upgrade-react-version
description: Upgrades React and ReactDOM to version 18.2.0 across all frontend repositories.

on:
  - repositoriesMatchingQuery: lang:typescript file:package.json org:sourcegraph

steps:
  - run: npm install react@18.2.0 react-dom@18.2.0
    container: node:18-alpine

  - run: npm run test
    container: node:18-alpine

changesetTemplate:
  title: Upgrade React to 18.2.0
  body: |
    This PR upgrades React and ReactDOM to version 18.2.0.
    
    Please review the changes and run manual tests if necessary.
  branch: batch-changes/upgrade-react-18
  commit: Upgrade React dependencies
  published: false # Do not publish PRs immediately

# Define transform rules for specific file types if needed
transform:
  - group:
    repository: github.com/sourcegraph/sourcegraph
